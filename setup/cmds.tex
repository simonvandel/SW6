% Point on the circumference on circle
\newcommand{\circoord}[5]{
  \FPeval\xcirpoint{#1 + #3 * cos(#4 * pi / 180)}
  \FPeval\ycirpoint{#2 + #3 * sin(#4 * pi / 180)}
  \coordinate (#5) at (\xcirpoint,\ycirpoint)
}

\newcommand{\circoordTwo}[5]{
  \FPeval\xcirpoint{#1 + #3 * cos(#4 * pi / 180)}
  \FPeval\ycirpoint{#2 + #3 * sin(#4 * pi / 180)}
  \coordinate (#5) at (axis cs:\xcirpoint,\ycirpoint)
}
